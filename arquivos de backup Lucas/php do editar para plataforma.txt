<select name="plataformas[]" multiple>
    <?php foreach ($todas_plataformas as $plataforma): ?>
        <?php
        // Procura diretamente dentro do resultado do banco se o ID existe lá
        // $plataformas_do_produto é o resultado direto do seu SELECT no banco
        $marcado = false;
        foreach ($plataformas_do_produto as $prod_plat) {
            if ($prod_plat['id_plataforma'] == $plataforma['id']) {
                $marcado = true;
                break;
            }
        }
        ?>
        <option value="<?= $plataforma['id']; ?>" <?= $marcado ? 'selected' : ''; ?>>
            <?= $plataforma['nome']; ?>
        </option>
    <?php endforeach; ?>
</select>
